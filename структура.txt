safar-cafe/                  ← корневая папка проекта

├── node_modules/            ← зависимости (express, sqlite3, bcrypt, express-session)

├── profil.html              ← страница профиля (пока пустая, создайте позже)
│
├── css/
│   ├── style.css            ← основные стили сайта
│   └── auth.css             ← стили только для login/register (glass-эффект)
│
├── js/
│   ├── script.js            ← основной скрипт (корзина, табы меню, мобильные оптимизации)
│   └── lang.js              ← мультиязычность (переводы)
│
└── images/                  ← все картинки меню и фона
